        clearInterval(donationFeedbackClock);
        resetDonationWaitCounter();

    $("#firstSlideNextButton").on("click", function () {
        //$("form[name='firstSlideForm']").$submitted = true

        let element = document.querySelector('[ng-controller="Ahbap"]');
        let ngController = angular.element(element);
        let ngScope = ngController.scope();
        let donationAmount = ngScope.DONATION_AMOUNT;

        if (ngScope.USER_DATA != undefined && ngScope.USER_DATA != null) {
            if (ngScope.USER_DATA.Email != null)
                ngScope.DONATION_MAIL = ngScope.USER_DATA.Email;
            if (ngScope.USER_DATA.Phone != null)
                ngScope.DONATION_PHONE = ngScope.USER_DATA.Phone.trim().replace(/[^0-9]/g, '');
            if (ngScope.USER_DATA.DisplayName != null)
                ngScope.DONATION_NAME_SURNAME = ngScope.USER_DATA.DisplayName;
        }
        ngScope.$apply()

        if (donationAmount >= "1" && donationAmount > 0) {
            nextStep();
        }
        else {
            if (donationAmount == null)
                $('#firstSlideMessageDivId').text("Donation Amount field is required, it can not be blank!");
            else if (donationAmount <= 0 || donationAmount < "1")
                $('#firstSlideMessageDivId').text("The amount must be greater than 0!");

            $('#firstSlideMessageDivId').show();
            setTimeout(function () { $('#firstSlideMessageDivId').hide() }, 3000);
        }

        donationFeedbackClock = setInterval(function () {
            donationWaitCounter += 1;
            if (donationWaitCounter > 1400) {
                $("#donationFeedbackModal").modal("toggle");
                clearDonationWaitCounter();
            }
        }, 1000);
    });

    function creditCardValidation(creditCradNum) {
        if (creditCradNum == undefined || creditCradNum == null || creditCradNum == "")
            return false;

        creditCradNum = creditCradNum.replace(/\s+/g, '');
        var regexMasterCard = /^5[1-5][0-9]{14}$|^2(?:2(?:2[1-9]|[3-9][0-9])|[3-6][0-9][0-9]|7(?:[01][0-9]|20))[0-9]{12}$/;
        var regexVisa = /^4[0-9]{12}(?:[0-9]{3})?$/;
        var regexAmericanExpress = /^3[47][0-9]{13}$/;
        var regexDiscover = /^65[4-9][0-9]{13}|64[4-9][0-9]{13}|6011[0-9]{12}|(622(?:12[6-9]|1[3-9][0-9]|[2-8][0-9][0-9]|9[01][0-9]|92[0-5])[0-9]{10})$/;
        var regexTroy = /^(?:9792|65\d{2}|36|2205)\d{12}$/;
        if (creditCradNum.match(regexMasterCard) || creditCradNum.match(regexVisa) || creditCradNum.match(regexAmericanExpress) || creditCradNum.match(regexDiscover) || creditCradNum.match(regexTroy))
            return true;
        else
            return false;
    }
    function creditCardFormat(ccid, ctid) {
        // supports Amex, Master Card, Visa, and Discover
        // parameter 1 ccid= id of credit card number field
        // parameter 2 ctid= id of credit card type field

        var ccNumString = document.getElementById(ccid).value;
        ccNumString = ccNumString.replace(/[^0-9]/g, '');
        // mc, starts with - 51 to 55
        // v, starts with - 4
        // dsc, starts with 6011, 622126-622925, 644-649, 65
        // amex, starts with 34 or 37
        var typeCheck = ccNumString.substring(0, 2);
        var cType = '';
        var block1 = '';
        var block2 = '';
        var block3 = '';
        var block4 = '';
        var formatted = '';

        if (typeCheck.length == 2) {
            typeCheck = parseInt(typeCheck);
            if (typeCheck >= 40 && typeCheck <= 49) {
                cType = 'Visa';
            }
            else if (typeCheck >= 51 && typeCheck <= 55) {
                cType = 'Master Card';
            }
            else if ((typeCheck >= 60 && typeCheck <= 62) || (typeCheck == 64) || (typeCheck == 65)) {
                cType = 'Discover';
            }
            else if (typeCheck == 34 || typeCheck == 37) {
                cType = 'American Express';
            }
            else if (typeCheck == 97) {
                cType = 'Troy';
            }
            else {
                cType = 'Invalid';
            }
        }

        // all support card types have a 4 digit firt block
        block1 = ccNumString.substring(0, 4);
        if (block1.length == 4) {
            block1 = block1 + ' ';
        }

        if (cType == 'Visa' || cType == 'Master Card' || cType == 'Discover' || cType == 'Troy') {
            // for 4X4 cards
            block2 = ccNumString.substring(4, 8);
            if (block2.length == 4) {
                block2 = block2 + ' ';
            }
            block3 = ccNumString.substring(8, 12);
            if (block3.length == 4) {
                block3 = block3 + ' ';
            }
            block4 = ccNumString.substring(12, 16);
        }
        else if (cType == 'American Express') {
            // for Amex cards
            block2 = ccNumString.substring(4, 10);
            if (block2.length == 6) {
                block2 = block2 + ' ';
            }
            block3 = ccNumString.substring(10, 15);
            block4 = '';
        }
        else if (cType == 'Invalid') {
            // for Amex cards
            block1 = typeCheck;
            block2 = '';
            block3 = '';
            block4 = '';

            $('#cardValidationMessageDivId').text("Invalid Card Number");
            $('#cardValidationMessageDivId').show();
            setTimeout(function () { $('#cardValidationMessageDivId').hide() }, 3000);
        }

        formatted = block1 + block2 + block3 + block4;
        document.getElementById(ccid).value = formatted;
        //document.getElementById(ctid).value=cType;
    }
